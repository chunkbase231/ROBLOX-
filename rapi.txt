function Main()
    SN = gg.choice({
        "ON",
        "OFF"
    }, nil, "fantasy sky")
    if SN == nil then
        return
    end

    -- 音效链接（假设@@@1是你后续要换的真实URL）
    local sound_url = "https://www.aconvert.com/view.php?l=s19.aconvert.com/convert/p3r68-cdx67/ew2t6-0ties.mp3"

    if SN == 1 then -- ON
        PlaySound(sound_url)
    elseif SN == 2 then -- OFF
        StopSound(sound_url)
    end
end

-- 播放指定链接音效函数（需要你的环境支持）
function PlaySound(url)
    -- 如果是 Roblox，假设 API 是 Sound 对象
    -- 伪代码如下，具体要用你的实际API:
    local sound = Instance.new("Sound")
    sound.SoundId = url
    sound.Parent = workspace
    sound:Play()
end

-- 停止音效函数
function StopSound(url)
    -- Roblox 示例：把播放的Sound停掉
    for _, obj in pairs(workspace:GetChildren()) do
        if obj:IsA("Sound") and obj.SoundId == url then
            obj:Stop()
            obj:Destroy()
        end
    end
end

-- 执行主函数
Main()
