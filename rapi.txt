function Main()
    SN = gg.choice({
        "ON",
        "OFF"
    }, 0, "fantasy sky")
    if SN == nil then
        os.exit()
    end

    -- 如果选择了 ON (返回值为 1)
    if SN == 1 then
        -- 导入安卓 MediaPlayer (使用 pcall 防止报错)
        pcall(function()
            import "android.media.MediaPlayer"
            mp = MediaPlayer()
            mp:setDataSource("https://www.aconvert.com/view.php?l=s19.aconvert.com/convert/p3r68-cdx67/ew2t6-0ties.mp3") -- 设置链接
            mp:prepareAsync()
            mp.setOnPreparedListener{onPrepared = function() mp:start() end}
            gg.toast("音效已开启")
        end)
    end

    -- 如果选择了 OFF (返回值为 2)
    if SN == 2 then
        -- 停止播放并释放资源
        pcall(function()
            if mp ~= nil then
                mp:stop()
                mp:release()
            end
            mp = nil
            gg.toast("音效已关闭")
        end)
        gg.setVisible(false) -- 顺便隐藏 GG 界面
    end
end

-- 全局变量 mp，确保在 Main 函数外也能被访问（用于停止）
mp = nil
Main() -- 运行主函数
